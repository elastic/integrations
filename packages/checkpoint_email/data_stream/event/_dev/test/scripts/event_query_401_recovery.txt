# Test that a 401 on the event query triggers re-authentication and recovery.
#
# The bug requires a prior successful cycle to cache the token in state.
# The auth result lives in a local variable; only the success handler copies
# access_token and expiry into the output state. So the test must see a 200
# before the 401 — otherwise expiry is never in state and auth reruns anyway.
#
# Mock design:
#   Auth endpoint: cycles [token-alpha, token-bravo].
#   Event query with token-alpha: cycles [200 with data-A, 401].
#   Event query with token-bravo: always 200 with data-B (different eventIds).
#
# With the fix (3 cycles):
#   1. auth(alpha) -> 200(data-A)  -> token cached in state
#   2. alpha cached -> 401         -> expiry cleared -> re-auth forced
#   3. auth(bravo) -> 200(data-B)  -> data-B events indexed
#
# Without the fix (alpha stays cached, bravo never reached):
#   1. auth(alpha) -> 200(data-A)  -> token cached in state
#   2. alpha cached -> 401         -> expiry NOT cleared
#   3. alpha cached -> 200(data-A) -> same eventIds, overwrites, no new docs
#   4. alpha cached -> 401         -> another error event
#   ... repeats, data-B events never appear.
#
# Assertion: data-B events exist. This can only happen via token-bravo.

[!external_stack] skip 'Skipping external stack test.'
[!exec:jq] skip 'Skipping test requiring absent jq command'

use_stack -profile ${CONFIG_PROFILES}/${PROFILE}
install_agent -profile ${CONFIG_PROFILES}/${PROFILE} -network_name NETWORK_NAME
docker_up -profile ${CONFIG_PROFILES}/${PROFILE} -network ${NETWORK_NAME} checkpoint-email-mock
add_package -profile ${CONFIG_PROFILES}/${PROFILE}
add_package_policy -profile ${CONFIG_PROFILES}/${PROFILE} test_config.yaml DATA_STREAM_NAME

# Wait for 5 documents: 2 data-A + 1 error + 2 data-B.
get_docs -profile ${CONFIG_PROFILES}/${PROFILE} -want 5 -timeout 5m ${DATA_STREAM_NAME}
cp stdout got_docs.json

# Verify at least one error event mentioning the 401 authentication failure.
exec jq -r '[.hits.hits[]._source.error.message // empty] | flatten | .[]' got_docs.json
stdout 'Authentication required'

# Verify data-B events exist. These can only be indexed if the integration
# re-authenticated after the 401 and obtained token-bravo.
exec jq '[.hits.hits[]._source | select(.event.id == "b6d8674a04c30123456789e4d3ebd98")] | length' got_docs.json
stdout '^1$'

# Clean up.
remove_package_policy -profile ${CONFIG_PROFILES}/${PROFILE} ${DATA_STREAM_NAME}
uninstall_agent -profile ${CONFIG_PROFILES}/${PROFILE} -timeout 1m
docker_down checkpoint-email-mock
