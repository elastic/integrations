# Test that the alert data stream recovers from a transient 403 Forbidden
# (Authorization_RequestDenied) response from the Graph Security API.
#
# The mock cycles through two responses for /v1.0/security/alerts_v2:
#   1st call → 403 {"error":{"code":"Authorization_RequestDenied",...}}
#   2nd call → 200 with 2 alert documents
#
# The httpjson input logs the 403, waits for the next interval, then retries.
# The test asserts that the 2 alert documents eventually land in Elasticsearch,
# confirming the integration recovers without operator intervention.
#
# Note: unlike CEL-based inputs, httpjson does not emit indexed error events
# for non-2xx responses. The observable outcome is that data collection
# resumes after the transient failure, not that an error document is indexed.

[!external_stack] skip 'Skipping external stack test.'
[!exec:jq] skip 'Skipping test requiring absent jq command'

# Connect to the running stack.
use_stack -profile ${CONFIG_PROFILES}/${PROFILE}

# Install an agent.
install_agent -profile ${CONFIG_PROFILES}/${PROFILE} -network_name NETWORK_NAME

# Start the mock Microsoft Graph Security API server.
docker_up -profile ${CONFIG_PROFILES}/${PROFILE} -network ${NETWORK_NAME} m365d-mock

# Add the package resources.
add_package -profile ${CONFIG_PROFILES}/${PROFILE}

# Add the data stream policy.
add_package_policy -profile ${CONFIG_PROFILES}/${PROFILE} test_config.yaml DATA_STREAM_NAME

# Wait for 2 alert documents. The first poll hits the 403; the integration
# retries on the next interval (30s) and the mock returns 200. The timeout
# is long enough to absorb the failed first attempt.
get_docs -profile ${CONFIG_PROFILES}/${PROFILE} -want 2 -timeout 5m ${DATA_STREAM_NAME}
cp stdout got_docs.json

# Verify the documents have the m365_defender.alert namespace set.
exec jq '[.hits.hits[]._source | select(.m365_defender.alert != null)] | length' got_docs.json
stdout '^2$'

# Clean up.
remove_package_policy -profile ${CONFIG_PROFILES}/${PROFILE} ${DATA_STREAM_NAME}
uninstall_agent -profile ${CONFIG_PROFILES}/${PROFILE} -timeout 1m
docker_down m365d-mock

-- test_config.yaml --
input: httpjson
vars:
  login_url: http://m365d-mock:8080
  client_id: test-client-id
  client_secret: test-client-secret
  tenant_id: test-tenant-id
data_stream:
  vars:
    request_url: http://m365d-mock:8080
    batch_size: 2
    interval: 30s
    initial_interval: 1h
    preserve_original_event: true
    enable_request_tracer: false
-- m365d-mock/docker-compose.yml --
version: '2.3'
services:
  m365d-mock:
    image: docker.elastic.co/observability/stream:v0.20.0
    hostname: m365d-mock
    ports:
      - 8080
    environment:
      PORT: "8080"
    volumes:
      - ./config.yml:/config.yml
    command:
      - http-server
      - --addr=:8080
      - --config=/config.yml
