# Test that a 401 Unauthorized response from the Trend Micro Vision One API
# is handled gracefully and collection recovers when the next request succeeds.
#
# httpjson logs the error and retries on the next interval. Recovery is the
# observable outcome: after the mock cycles to the 200 response, 2 detection
# documents are indexed.

[!external_stack] skip 'Skipping external stack test.'
[!exec:jq] skip 'Skipping test requiring absent jq command'

# Connect to the running stack.
use_stack -profile ${CONFIG_PROFILES}/${PROFILE}

# Install an agent.
install_agent -profile ${CONFIG_PROFILES}/${PROFILE} -network_name NETWORK_NAME

# Start the mock Trend Micro Vision One API server.
docker_up -profile ${CONFIG_PROFILES}/${PROFILE} -network ${NETWORK_NAME} tmv1-mock

# Add the package resources.
add_package -profile ${CONFIG_PROFILES}/${PROFILE}

# Add the data stream policy pointing at the mock server.
add_package_policy -profile ${CONFIG_PROFILES}/${PROFILE} test_config.yaml DATA_STREAM_NAME

# Wait for 2 detection documents (from the 200 response after the mock cycles
# past the 401). The 401 response causes httpjson to log an error and retry on
# the next interval.
get_docs -profile ${CONFIG_PROFILES}/${PROFILE} -want 2 -timeout 5m ${DATA_STREAM_NAME}
cp stdout got_docs.json

# Verify the documents have the trend_micro_vision_one.detection namespace set.
exec jq '[.hits.hits[]._source | select(.trend_micro_vision_one.detection != null)] | length' got_docs.json
stdout '^2$'

# Clean up.
remove_package_policy -profile ${CONFIG_PROFILES}/${PROFILE} ${DATA_STREAM_NAME}
uninstall_agent -profile ${CONFIG_PROFILES}/${PROFILE} -timeout 1m
docker_down tmv1-mock

-- test_config.yaml --
input: httpjson
vars:
  hostname: http://tmv1-mock:8080
  api_token: test-api-token
data_stream:
  vars:
    interval: 30s
    initial_interval: 1h
    preserve_original_event: true
    enable_request_tracer: false
-- tmv1-mock/docker-compose.yml --
version: '2.3'
services:
  tmv1-mock:
    image: docker.elastic.co/observability/stream:v0.20.0
    hostname: tmv1-mock
    ports:
      - 8080
    environment:
      PORT: "8080"
    volumes:
      - ./config.yml:/config.yml
    command:
      - http-server
      - --addr=:8080
      - --config=/config.yml
-- tmv1-mock/config.yml --
rules:
  # Detections endpoint: first request returns 401 (invalid token), second returns 2 detections.
  # The mock cycles through responses in order. The test completes after the
  # second request and docker_down stops the mock before it wraps back.
  - path: /v3.0/search/detections
    methods: [GET]
    responses:
      - status_code: 401
        headers:
          Content-Type:
            - "application/json"
        body: |-
          {"message":"Unauthorized","code":401}
      - status_code: 200
        headers:
          Content-Type:
            - "application/json"
        body: |-
          {"items":[{"uuid":"detection-001","eventTime":1705312800000,"detectionName":"Test Detection 1","pname":"svchost.exe","pfilePath":"C:\\Windows\\System32\\svchost.exe"},{"uuid":"detection-002","eventTime":1705312860000,"detectionName":"Test Detection 2","pname":"cmd.exe","pfilePath":"C:\\Windows\\System32\\cmd.exe"}]}
