# Test that configuring an invalid Auth-Key results in an error event and
# no url data is collected.
#
# The mock accepts requests with the valid Auth-Key header and returns 403
# for all other requests. The CEL program treats non-200 responses as errors
# and emits a single error event with want_more=false, stopping collection.

[!external_stack] skip 'Skipping external stack test.'
[!exec:jq] skip 'Skipping test requiring absent jq command'

# Connect to the running stack.
use_stack -profile ${CONFIG_PROFILES}/${PROFILE}

# Install an agent.
install_agent -profile ${CONFIG_PROFILES}/${PROFILE} -network_name NETWORK_NAME

# Start the mock AbuseCH API server.
docker_up -profile ${CONFIG_PROFILES}/${PROFILE} -network ${NETWORK_NAME} abusech-mock

# Add the package resources.
add_package -profile ${CONFIG_PROFILES}/${PROFILE}

# Add the data stream policy with an invalid auth key.
add_package_policy -profile ${CONFIG_PROFILES}/${PROFILE} test_config.yaml DATA_STREAM_NAME

# Wait for exactly 1 document (the error event). Use a longer confirm duration
# to verify no additional data events arrive.
get_docs -profile ${CONFIG_PROFILES}/${PROFILE} -want 1 -confirm 15s -timeout 5m ${DATA_STREAM_NAME}
cp stdout got_docs.json

# The single document should be an error event.
exec jq '[.hits.hits[]._source | select(.error != null)] | length' got_docs.json
stdout '^1$'

# No url data events should be present.
exec jq '[.hits.hits[]._source | select(.abusech.url != null)] | length' got_docs.json
stdout '^0$'

# Clean up.
remove_package_policy -profile ${CONFIG_PROFILES}/${PROFILE} ${DATA_STREAM_NAME}
uninstall_agent -profile ${CONFIG_PROFILES}/${PROFILE} -timeout 1m
docker_down abusech-mock

-- test_config.yaml --
input: cel
vars:
  auth_key: wrong-auth-key
data_stream:
  vars:
    url: http://abusech-mock:8080/api/v1/downloads/
    interval: 30s
    preserve_original_event: true
    enable_request_tracer: false
-- abusech-mock/docker-compose.yml --
version: '2.3'
services:
  abusech-mock:
    image: docker.elastic.co/observability/stream:v0.20.0
    hostname: abusech-mock
    ports:
      - 8080
    environment:
      PORT: "8080"
    volumes:
      - ./config.yml:/config.yml
    command:
      - http-server
      - --addr=:8080
      - --config=/config.yml
-- abusech-mock/config.yml --
rules:
  # Requests with the valid auth key succeed.
  - path: /api/v1/downloads/
    methods: [GET]
    request_headers:
      Auth-Key:
        - "valid-auth-key"
    responses:
      - status_code: 200
        headers:
          Content-Type:
            - "application/zip"
        body: |-
          {}

  # Any other request (wrong or missing auth key) returns 403.
  - path: /api/v1/downloads/
    methods: [GET]
    responses:
      - status_code: 403
        headers:
          Content-Type:
            - "application/json"
        body: |-
          {"query_status":"auth_key_required","data":null}
