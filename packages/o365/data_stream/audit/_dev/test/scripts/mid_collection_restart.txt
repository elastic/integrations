# Test that restarting the agent mid-collection does not lose events.
#
# The mock API serves two content types (Audit.SharePoint and Audit.General),
# each with two content blobs. After the first collection cycle produces 4
# events, the agent is stopped (simulating a pod death mid-collection) and
# restarted. On resume, the mock continues to return the same blobs.
#
# The test verifies:
# 1. All 4 original events survive the restart (deduplicated by _id/fingerprint).
# 2. Post-restart collection produces 2 new events (from newly available blobs).
# 3. Total event count is 6 (no skipped windows, no lost data).

[!external_stack] skip 'Skipping external stack test.'
[!exec:jq] skip 'Skipping test requiring absent jq command'

# Connect to the running stack.
use_stack -profile ${CONFIG_PROFILES}/${PROFILE}

# Install an agent.
install_agent -profile ${CONFIG_PROFILES}/${PROFILE} -network_name NETWORK_NAME

# Start mock O365 API (phase 1: two blobs per content type).
docker_up -profile ${CONFIG_PROFILES}/${PROFILE} -network ${NETWORK_NAME} o365-mock

# Install the dev package.
add_package -profile ${CONFIG_PROFILES}/${PROFILE}

# Create policy and start collection.
add_package_policy -profile ${CONFIG_PROFILES}/${PROFILE} test_config.yaml DATA_STREAM_NAME

# Wait for 4 events: 2 blobs x 2 content types.
get_docs -profile ${CONFIG_PROFILES}/${PROFILE} -want 4 -confirm 15s -timeout 5m ${DATA_STREAM_NAME}
cp stdout got_docs_before.json

# Verify all 4 distinct event IDs are present.
exec jq '[.hits.hits[]._source.o365.audit.Id] | sort | join(",")' got_docs_before.json
stdout '"gen-event-001,gen-event-002,sp-event-001,sp-event-002"'

# --- Simulate mid-collection restart (pod death) ---
# Stop and restart the agent. The cursor is persisted by Fleet, so on
# restart the CEL program resumes from cursor.last_for and cursor.todo_content.
uninstall_agent -profile ${CONFIG_PROFILES}/${PROFILE} -timeout 1m

# Swap mock to phase 2: original blobs still available, plus new blobs.
docker_down o365-mock
cp o365-mock/config-phase2.yml o365-mock/config.yml
docker_up -profile ${CONFIG_PROFILES}/${PROFILE} -network ${NETWORK_NAME} o365-mock

# Reinstall agent (simulates pod restart with fresh agent connecting to Fleet).
install_agent -profile ${CONFIG_PROFILES}/${PROFILE} -network_name NETWORK_NAME

# Wait for 6 total events: 4 original (deduplicated) + 2 new.
get_docs -profile ${CONFIG_PROFILES}/${PROFILE} -want 6 -confirm 15s -timeout 5m ${DATA_STREAM_NAME}
cp stdout got_docs_after.json

# Verify all 6 distinct event IDs are present.
exec jq '[.hits.hits[]._source.o365.audit.Id] | sort | join(",")' got_docs_after.json
stdout '"gen-event-001,gen-event-002,gen-event-003,sp-event-001,sp-event-002,sp-event-003"'

# Verify no duplicate documents indexed (fingerprint dedup on o365audit.Id).
exec jq '.hits.hits | length' got_docs_after.json
stdout '^6$'

# Clean up.
remove_package_policy -profile ${CONFIG_PROFILES}/${PROFILE} ${DATA_STREAM_NAME}
uninstall_agent -profile ${CONFIG_PROFILES}/${PROFILE} -timeout 1m
docker_down o365-mock
