# Test that replacing the agent (full re-enrollment, no cursor carried over)
# does not lose events.
#
# The mock API serves two content types (Audit.SharePoint and Audit.General),
# each with three content blobs. After the first collection cycle produces 6
# events, the agent is destroyed and a new one is enrolled from scratch.
# The new agent has no cursor state -- it re-collects from initial_interval.
# The mock is swapped to a different set of blobs: event 001 is removed
# (simulating expired/rotated content) and event 004 is added (new content
# that appeared during the outage).
#
# Phase 1: {sp,gen}-event-{001,002,003} → 6 events
# Phase 2: {sp,gen}-event-{002,003,004} → 6 events (001 gone, 004 new)
#
# Each install_agent creates a new Fleet policy with a unique namespace, so
# each phase writes to a separate data stream. Phase 1 events live in one
# data stream; phase 2 events live in another. The test verifies each phase
# independently and then confirms all 8 distinct IDs exist across both.
#
# NOTE: This does NOT test mid-drain cursor consistency (cursor.last_for
# advancing ahead of todo_content). That scenario requires killing the
# agent container while blobs are still queued and preserving the cursor
# across restart -- which the script test infrastructure does not support
# (uninstall_agent destroys the enrollment and cursor).

[!external_stack] skip 'Skipping external stack test.'
[!exec:jq] skip 'Skipping test requiring absent jq command'

# Connect to the running stack.
use_stack -profile ${CONFIG_PROFILES}/${PROFILE}

# Install an agent.
install_agent -profile ${CONFIG_PROFILES}/${PROFILE} -network_name NETWORK_NAME

# Start mock O365 API (phase 1: three blobs per content type).
docker_up -profile ${CONFIG_PROFILES}/${PROFILE} -network ${NETWORK_NAME} o365-mock

# Install the dev package.
add_package -profile ${CONFIG_PROFILES}/${PROFILE}

# Create policy and start collection.
add_package_policy -profile ${CONFIG_PROFILES}/${PROFILE} test_config.yaml DATA_STREAM_NAME

# Wait for 6 events: 3 blobs x 2 content types.
get_docs -profile ${CONFIG_PROFILES}/${PROFILE} -want 6 -confirm 15s -timeout 5m ${DATA_STREAM_NAME}
cp stdout got_docs_phase1.json

# Verify all 6 distinct event IDs are present.
exec jq '[.hits.hits[]._source.event.id] | sort | join(",")' got_docs_phase1.json
stdout '"gen-event-001,gen-event-002,gen-event-003,sp-event-001,sp-event-002,sp-event-003"'

# --- Simulate agent replacement (node rotation, pod eviction) ---
# Destroy the agent entirely. The new agent will have no cursor state
# and will re-collect from initial_interval.
uninstall_agent -profile ${CONFIG_PROFILES}/${PROFILE} -timeout 1m
docker_down o365-mock

# Swap mock to phase 2: blob 001 removed, blob 004 added.
cp o365-mock/config-phase2.yml o365-mock/config.yml

# Install agent first so NETWORK_NAME is refreshed, then start the mock
# on the agent's network so DNS resolution works.
install_agent -profile ${CONFIG_PROFILES}/${PROFILE} -network_name NETWORK_NAME
docker_up -profile ${CONFIG_PROFILES}/${PROFILE} -network ${NETWORK_NAME} o365-mock

# Re-assign the package policy. This sets DATA_STREAM_NAME to the new
# data stream (different namespace from phase 1).
add_package_policy -profile ${CONFIG_PROFILES}/${PROFILE} test_config.yaml DATA_STREAM_NAME

# Wait for 6 phase 2 events in the new data stream.
get_docs -profile ${CONFIG_PROFILES}/${PROFILE} -want 6 -confirm 15s -timeout 5m ${DATA_STREAM_NAME}
cp stdout got_docs_phase2.json

# Phase 2 must contain events 002, 003, 004 (re-collected overlap + new).
exec jq '[.hits.hits[]._source.event.id] | sort | join(",")' got_docs_phase2.json
stdout '"gen-event-002,gen-event-003,gen-event-004,sp-event-002,sp-event-003,sp-event-004"'

# Combine both phases and verify all 8 distinct event IDs are present:
# 001 preserved from phase 1, 002/003 overlap, 004 new from phase 2.
exec jq -s '[.[].hits.hits[]._source.event.id] | unique | sort | join(",")' got_docs_phase1.json got_docs_phase2.json
stdout '"gen-event-001,gen-event-002,gen-event-003,gen-event-004,sp-event-001,sp-event-002,sp-event-003,sp-event-004"'

# Clean up.
remove_package_policy -profile ${CONFIG_PROFILES}/${PROFILE} ${DATA_STREAM_NAME}
uninstall_agent -profile ${CONFIG_PROFILES}/${PROFILE} -timeout 1m
docker_down o365-mock
