# Test that configuring invalid API credentials results in an error event
# and no intelligence data is collected.
#
# The mock returns 401 Unauthorized for any request that does not carry the
# valid Authorization header. The CEL program treats non-200 responses as
# errors, emits a single error event, and sets want_more=false.

[!external_stack] skip 'Skipping external stack test.'
[!exec:jq] skip 'Skipping test requiring absent jq command'

# Connect to the running stack.
use_stack -profile ${CONFIG_PROFILES}/${PROFILE}

# Install an agent.
install_agent -profile ${CONFIG_PROFILES}/${PROFILE} -network_name NETWORK_NAME

# Start the mock Anomali ThreatStream API server.
docker_up -profile ${CONFIG_PROFILES}/${PROFILE} -network ${NETWORK_NAME} anomali-mock

# Add the package resources.
add_package -profile ${CONFIG_PROFILES}/${PROFILE}

# Add the data stream policy with invalid credentials.
add_package_policy -profile ${CONFIG_PROFILES}/${PROFILE} test_config.yaml DATA_STREAM_NAME

# Wait for exactly 1 document (the error event). Use a longer confirm duration
# to verify no additional data events arrive.
get_docs -profile ${CONFIG_PROFILES}/${PROFILE} -want 1 -confirm 15s -timeout 5m ${DATA_STREAM_NAME}
cp stdout got_docs.json

# The single document should be an error event referencing the 401 response.
exec jq -r '[.hits.hits[]._source.error.message // empty] | flatten | .[]' got_docs.json
stdout '401'

# No intelligence data events should be present.
exec jq '[.hits.hits[]._source | select(.anomali.threatstream != null)] | length' got_docs.json
stdout '^0$'

# Clean up.
remove_package_policy -profile ${CONFIG_PROFILES}/${PROFILE} ${DATA_STREAM_NAME}
uninstall_agent -profile ${CONFIG_PROFILES}/${PROFILE} -timeout 1m
docker_down anomali-mock

-- test_config.yaml --
input: cel
vars: ~
data_stream:
  vars:
    username: test-user
    api_key: wrong-api-key
    url: http://anomali-mock:8080
    interval: 30s
    initial_interval: 24h
    page_size: 10
    ioc_duration_before_deletion: 90d
    preserve_original_event: true
    enable_request_tracer: false
-- anomali-mock/docker-compose.yml --
version: '2.3'
services:
  anomali-mock:
    image: docker.elastic.co/observability/stream:v0.20.0
    hostname: anomali-mock
    ports:
      - 8080
    environment:
      PORT: "8080"
    volumes:
      - ./config.yml:/config.yml
    command:
      - http-server
      - --addr=:8080
      - --config=/config.yml
-- anomali-mock/config.yml --
rules:
  # Requests with valid credentials succeed.
  - path: /api/v2/intelligence/
    methods: [GET]
    request_headers:
      Authorization:
        - "apikey valid-user:valid-api-key"
    responses:
      - status_code: 200
        headers:
          Content-Type:
            - "application/json"
        body: |-
          {"objects":[{"update_id":100000001,"type":"ip","itype":"c2_ip","value":"1.2.3.4","confidence":75,"threat_type":"c2","status":"active","is_public":true,"created_ts":"2024-01-15T10:00:00Z","modified_ts":"2024-01-15T10:00:00Z","source":"Test"}],"meta":{"next":null,"total_count":1,"offset":0,"limit":10}}

  # Any other request (wrong credentials) returns 401.
  - path: /api/v2/intelligence/
    methods: [GET]
    responses:
      - status_code: 401
        headers:
          Content-Type:
            - "application/json"
        body: |-
          {"error_code":401,"error":"Unauthorized","message":"Invalid credentials provided"}
