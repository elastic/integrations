# Juniper integration

This is an integration for ingesting logs from [Juniper NetScreen](https://www.juniper.net/documentation/en_US/release-independent/screenos/information-products/pathway-pages/netscreen-series/product/).

### Log

The `log` dataset collects Netscreen logs.

An example event for `log` looks as following:

```json
{
    "@timestamp": "2016-01-29T06:09:59.000Z",
    "agent": {
        "ephemeral_id": "c439baa6-c1f5-4533-bb61-3a020bd4e4f9",
        "id": "11de7269-3d5a-4523-8b1f-e40ea1e2be97",
        "name": "docker-fleet-agent",
        "type": "filebeat",
        "version": "8.4.1"
    },
    "data_stream": {
        "dataset": "juniper_netscreen.log",
        "namespace": "ep",
        "type": "logs"
    },
    "ecs": {
        "version": "8.8.0"
    },
    "elastic_agent": {
        "id": "11de7269-3d5a-4523-8b1f-e40ea1e2be97",
        "snapshot": false,
        "version": "8.4.1"
    },
    "event": {
        "agent_id_status": "verified",
        "code": "00628",
        "dataset": "juniper_netscreen.log",
        "ingested": "2022-09-28T06:19:48Z",
        "timezone": "+00:00"
    },
    "input": {
        "type": "udp"
    },
    "log": {
        "level": "low",
        "source": {
            "address": "192.168.176.4:38747"
        }
    },
    "observer": {
        "product": "Netscreen",
        "type": "Firewall",
        "vendor": "Juniper"
    },
    "rsa": {
        "internal": {
            "messageid": "00628"
        },
        "misc": {
            "hardware_id": "olab",
            "severity": "low"
        },
        "time": {
            "event_time": "2016-01-29T06:09:59.000Z"
        }
    },
    "tags": [
        "juniper-netscreen",
        "forwarded"
    ]
}
```
